[Event]
-Name:NONE_Script_prov
-Priority:-1
-Prq.IsGlobalEvent:0
-Script.In:
	actor_SetActionLabel(thisActor,"ACTOR.1");
	//промежуточный ивент с проверками
	actor_OrientHead(thisActor,90);
	setGlobal("rest_time",1);
	actorWait_Timeout(getGlobal("rest_time"));
	actor_OrientHead(thisActor,270);
	actorWait_Timeout(1);
	
	//чтобы усталость не стала отрицательной...потом это зафигаивается в глобальную переменную и рулится
	if (actor_GetAttributeN(thisActor,"tire")<getGlobal("step_of_rest")){
		actor_SetAttribute(thisActor,"tire",getGlobal("step_of_rest"));
	}
	//вывод актора из цикла...здесь тоже нужна та же глобальная переменная
	actor_SetAttribute(thisActor,"tire",actor_GetAttributeN(thisActor,"tire")-getGlobal("step_of_rest"));
	actor_AddIcon(thisActor, "boredom_m35");
	
	if (actor_GetAttributeN(thisActor,"tire")==0){
		actor_SwitchToAction(thisActor,"");
		actor_AddIcon(thisActor, "full_rest");
		return;
	}
	eventManager(thisActor,"NONE_Script*");
